  8            kp_knee, kd_knee, kp_ankle, kd_ankle, ankle_des_in, knee_des_in,...
  9            vel_filter_coeff, IMU_filter_coeff,...
 10            SLIP_ON, lt, k, d, L0, ...
 11            KPBC_ON, pbc_gain_ankle, pbc_gain_knee, M, Eref,...
 12            knee_stop_low, knee_stop_high, ankle_stop_low, ankle_stop_high,...
 13            max_torque,...
 14            hip_thresh_st, hip_vel_thresh_st, F_thresh,...
 15            Command_State,...
 16            KPBC_max_torque, KPBC_joint_level)

 19
 20% Knee_joint_position, Ankle_joint_position are in degrees, in a
 21% "biomechanics frame"
 22
 23% t, dt, t_out are in seconds
 24
 25    %% Persistent Variable Definitions
 26%Persistent variables used to store data between iterations
 27persistent hip_pos_prev...
 28           knee_pos_prev...
 29           ankle_pos_prev...
 30           knee_vel_prev...
 31           ankle_vel_prev...
 32           hip_vel_prev...
 33           IMU_pitch_prev...
 34           Esys_integrate...
 35           t_switched_phase...
 36           Stance...
 37           Swing...
 38           ForceCount...
 39           Edis ...
 40           Hip_Pos_Max...
 41           Hip_Pos_Min...
 42           IMU_LIVE...
 43           PushOff...
 44           %t_switched_swing... %implement swing setpoint switching hold?
 45       
 46    if isempty(knee_pos_prev)
 47        hip_pos_prev = IMU_pitch;
 48        knee_pos_prev = 0;
 49        ankle_pos_prev = 0;
 50        knee_vel_prev = 0;
 51        ankle_vel_prev = 0;
 52        hip_vel_prev = 0;
 53        IMU_pitch_prev = 0;
 54        Esys_integrate  = 0;
 55        t_switched_phase = time_in;
 56        %t_switched_swing = time_in;
 57        Stance = true;
 58        Swing = false;
 59        ForceCount = 0;
 60        Edis = 0;
 61        Hip_Pos_Max  = 23;
 62        Hip_Pos_Min = -20;
 63        IMU_LIVE = true;
 64        PushOff = false;
 65    end
 66
 67    %% Initialization and Hard Coded Values
 68    %Knee_torque_command = 0; 
 69    %Ankle_torque_command = 0; 
 70    %deltaL = single(0); %meters
 71    %Esys = single(0); %Joules
 72    %g = single(9.81); %m/s^2
 73    %pulse_length = single(10); %seconds
 74    %hip_pos = 0; 
 75    %t_out = 0; 
 76    %dt = 0;
 77    Mt = M;
 78    Ms = single(6.991429/2.205); % converted to kg
 79    Mf = single(0.55 + 1.643130/2.205); %carbon fiber foot + mechanism, kg
 80    %lt = single(0.3733); %meters
 81    ls = single(0.3733); %meters
 82    la = single(0.0628); %meters
 83    lf = single(0.15); %meters
 84    px = single(0); %meters
 85    py = single(0); %meters
 86    params = [Mt, Ms, Mf, lt, ls, la, lf, px, py]; %From ordering in makeMatlabFunctionsProthesisTestBench
 87    
 88    Knee_torque_command= single(0);
 89    Ankle_torque_command = single(0);
 90    knee_des_out = nan;
 91    ankle_des_out = nan;
 92    
 93    U_LIN_DAMP_K = single(0);
 94    U_LIN_DAMP_A = single(0);
 95    U_LIN_SPRING_K = single(0);
 96    U_LIN_SPRING_A = single(0);
 97    U_PBC_K = single(0);
 98    U_PBC_A = single(0);
 99    phase_var_out = single(0);
100    
101    % Position/velocity hard limits
102    ANKLE_POS_MIN_LIM  = -35;
103    ANKLE_POS_MAX_LIM = 35;
104    ANKLE_VEL_MIN_LIM = -200; %deg/s
105    ANKLE_VEL_MAX_LIM = 200; %deg/s
106    KNEE_POS_MIN_LIM = 2;
107    KNEE_POS_MAX_LIM = 105;
108    KNEE_VEL_MIN_LIM = -400; %deg/s
109    KNEE_VEL_MAX_LIM = 400; %deg/s
110
111    Edismax = 50; %joules
112    
113    %Winters Data Arrays
114    a_knee=[-21.9472500000000,4.24525000000000,21.6720000000000,0;20.9295000000000,-6.97650000000000,7.71900000000000,0;15.9362500000000,-0.998250000000000,18.6640000000000,25.8830000000000;-59.8412500000000,13.6422500000000,64.8630000000000,0;-19.4405000000000,-1.40250000000000,38.4100000000000,-47.2960000000000;45.1070000000000,-7.15300000000000,0.456000000000000,0;1.42525000000000,0.0657500000000000,2.21000000000000,3.24500000000000];
115    a_ankle=[5.50050000000000,-0.880500000000000,-4.60000000000000,0;-8.98500000000000,3.77020000000000,5.26600000000000,4.65120000000000;0.812500000000000,-0.475000000000000,7.74100000000000,2.81250000000000;-2.19975000000000,0.320750000000000,9.62000000000000,0;-14.7575000000000,0.878500000000001,-0.745000000000000,-24.2440000000000;23.0932000000000,-3.91420000000000,-19.9240000000000,0;-29.8020000000000,9.93400000000000,-0.0560000000000000,0;0.703500000000000,-0.234500000000000,-0.525000000000000,0;-2.60850000000000,0.869500000000000,1.21400000000000,0;-0.729000000000000,0.0190000000000000,0.580000000000000,-1.34400000000000];
116    knee_ind=[1 139 400 530 720 848 976 1001];
117    ankle_ind=[1 61 205 330 440 550 653 846 898 977 1001];
118    
119    %% State Calculations
120    %Assign joint positions
121    hip_pos = (1-IMU_filter_coeff)*IMU_pitch_prev + IMU_filter_coeff*IMU_pitch;
122    knee_pos =  Knee_joint_position; 
123    ankle_pos = Ankle_joint_position;
124    
125    %Calculate joint velocities usindg exponential smoothing filter
126    %https://en.wikipedia.org/wiki/Exponential_smoothing
127    hip_vel = (1-vel_filter_coeff)*hip_vel_prev + vel_filter_coeff*(hip_pos-hip_pos_prev)/dt;
128    knee_vel = (1-vel_filter_coeff)*knee_vel_prev + vel_filter_coeff*(knee_pos-knee_pos_prev)/dt;
129    ankle_vel = (1-vel_filter_coeff)*ankle_vel_prev + vel_filter_coeff*(ankle_pos-ankle_pos_prev)/dt;
130    
131    %Human Leg as a Robot states
132    x = zeros(10,1); %x,y,-knee ankle, ankle angle
133    x(4) =  -knee_pos; %reverse knee sign convention of biomechanics versus biped modeling     
134    x(5) =  ankle_pos;
135
136    x(9) =  -knee_vel; %reverse knee sign convention of biomechanics versus biped modeling
137    x(10) =  ankle_vel;
138   
139    %Virtual linear spring from hip to foot, l1 is shank length, l2 is thigh length     
140    J = Spring_Jacobian_func(x,params);
141    L = Spring_Length_func(x,params);
142    deltaL = L - L0;
143    Ldot = Spring_vel_func(x,params);
144    
145    %Calculate system energy
146    %Ehip = 1/2*M*(Ldot^2) + (L-L0)*M*g; %kinetic and potential energy of slip model
147    Espring = 1/2*k*deltaL^2; %virtual spring potential energy
148    KE = 1/2*M*(Ldot^2);%KE_func(x,params);
149    PE = PE_func(x,params);
150    Esys = Espring + KE;%Espring + PE + KE;
151    
152    %Phase Variable 
153    %s_a=clamp(1+(1-s_m)/(q_h_0-q_h_m)*(hip-q_h_0),0,1);
154    phase_var_out =  0.6 + 0.4*(hip_pos-Hip_Pos_Min)/(Hip_Pos_Max-Hip_Pos_Min);
155    phase_var_out = clamp(phase_var_out,0.6,1);
156    
157    %IMU State check
158    if abs(hip_pos_prev-hip_pos)>1
159        IMU_LIVE = false;
160    end
161    
162    PushOff = and(or((hip_pos < hip_thresh_st),PushOff),Stance); %PushOff is persistent, so this remembers when the hip_pos crosses and stays high until Stance is off
163    
164    %% Control logic
165    %Determine foot contact
166    if norm([Load_cell_x_force, Load_cell_y_force, Load_cell_z_force],2) > F_thresh
167        ForceCount = ForceCount+1;
168        if ForceCount > 2
169            FootContact = true;
170        else
171            FootContact = false;
172        end
173    else
174        FootContact = false;
175        ForceCount = 0;   
176    end
177             
178    StanceGain = 0;
179    SwingGain = 0;
180    t_hold = 0.1;
181    
182    %Phase State management, set autodetection or command stance/swing
183    if int8(Command_State) == int8(0)
184        %Determine stance vs swing, only allow phase state switching every t_hold seconds
185        if Stance && ~FootContact %The state needs to be switched
186            if t_hold < (time_in - t_switched_phase) %Check the last time we switched, if its too fast, wait
187               %Switch to SWING
188               t_switched_phase = time_in;
189               Swing = true;
190               Stance = false;
191               %reset the energy intergration variables
192               Esys_integrate = 0;
193               Edis = 0; 
194            end
195        elseif Swing && FootContact %ditto, but for the other case
196            if t_hold < (time_in - t_switched_phase) 
197                %Switch to STANCE
198                t_switched_phase = time_in;
199                Swing = false;
200                Stance = true;
201            end
202        end
203        %Manage torque smoothing when switching phases
204        if Stance
205            t_norm = (time_in - t_switched_phase)/(t_hold*0.8);
206            StanceGain = min(t_norm,1);
207            SwingGain = 1-StanceGain; 
208        elseif Swing
209            t_norm = (time_in - t_switched_phase)/(t_hold*0.8);
210            SwingGain = min(t_norm,1);
211            StanceGain = 1-SwingGain; 
212        end
213    elseif int8(Command_State) == int8(1) %Stance
214        Stance = true;
215        Swing = false;
216        Hip_Pos_Max = 23;
217        Hip_Pos_Min = - 20;
218        StanceGain = 1;
219        SwingGain = 0;
220    elseif int8(Command_State) == int8(2) %Swing
221        Stance = false;
222        Swing = true;
223        Hip_Pos_Max = 23;
224        Hip_Pos_Min = - 20;
225        StanceGain = 0;
226        SwingGain = 1;
227    end
228
229    %Torque computation
230    if SLIP_ON       
231        Knee_torque_command_stance = 0;
232        Ankle_torque_command_stance = 0;
233        %% Stance                
234        %Virtual linear spring
235        u_lin_spring = -k.*(deltaL).*J;     
236        u_lin_damp = -d.*(Ldot).*J;
237
238        U_LIN_DAMP_K = -u_lin_damp(4);% -u_G_shape(4);
239        U_LIN_DAMP_A = u_lin_damp(5);% u_G_shape(5);
240        U_LIN_SPRING_K = -u_lin_spring(4);
241        U_LIN_SPRING_A = u_lin_spring(5);
242
243        Knee_torque_command_stance  = Knee_torque_command_stance + U_LIN_DAMP_K + U_LIN_SPRING_K; %reverse knee sign convention of biomechanics versus biped modeling
244        Ankle_torque_command_stance = Ankle_torque_command_stance + U_LIN_DAMP_A + U_LIN_SPRING_A;
245
246        %estimate how much energy the virtual linear damper is dissipating, to be added back in during pushoff
247        if ~PushOff
248            Edis = Edis  + (knee_vel*u_lin_damp(4) + ankle_vel*u_lin_damp(5))*dt;
249        end
250
251        if PushOff && KPBC_ON %Also use energy tracking controller   
252            u_pbc = single(zeros(5,1));
253            
254            if KPBC_joint_level 
255                %Joint level KPBC
256                u_pbc(4) = -pbc_gain_knee * knee_vel * (Esys_integrate - Eref - max(Edis,Edismax));
257                u_pbc(5) = -pbc_gain_ankle * ankle_vel * (Esys_integrate - Eref - max(Edis,Edismax));
258                
259                U_PBC_K = u_pbc(4); %THE SIGN NEEDS TO BE CHANGED IF USING SPRING VERSUS JOINT LEVEL BECAUSE OF BIO VS ROBOT KNEE CORDS
260                U_PBC_A = u_pbc(5);
261                
262            else
263                %KPBC along spring axis
264                u_pbc  = -J*pbc_gain_knee*(Esys - Eref)*Ldot; 
265                
266                U_PBC_K = -u_pbc(4); %THE SIGN NEEDS TO BE CHANGED IF USING SPRING VERSUS JOINT LEVEL BECAUSE OF BIO VS ROBOT KNEE CORDS
267                U_PBC_A = u_pbc(5);
268                
269            end
270                                          
271            %Biomemetic power saturation
272            Pow_Knee = knee_vel*U_PBC_K ;
273            Pow_Ankle = ankle_vel*U_PBC_A;
274            if sign(Pow_Ankle) == -1
275                U_PBC_A = 0;
276            end
277            if sign(Pow_Knee) == 1
278                U_PBC_K = 0;
279            end
280            
281            %Absolute saturation
282            if KPBC_max_torque>0
283                U_PBC_K = Saturate(U_PBC_K,-KPBC_max_torque,KPBC_max_torque);
284                U_PBC_A = Saturate(U_PBC_A,-KPBC_max_torque,KPBC_max_torque);
285            end
286            
287            %Use energy intergration scheme  
288            Esys_integrate = Esys_integrate  + (knee_vel*U_PBC_K + ankle_vel*U_PBC_A)*dt;
289                     
290            %cancel damping term during KPBC
291            Knee_torque_command_stance = Knee_torque_command_stance + U_PBC_K - U_LIN_DAMP_K;
292            Ankle_torque_command_stance = Ankle_torque_command_stance + U_PBC_A - U_LIN_DAMP_A;
293        end
294        %% Swing
295        if IMU_LIVE
296            %Use hip pos as phase variable to index winters data
297            [knee_des_out ,~,ankle_des_out,~]=winterFit(phase_var_out*1000,a_knee,a_ankle,knee_ind,ankle_ind);
298
299            %follow the trjactory via PD controller
300            [Knee_torque_command_swing,Ankle_torque_command_swing] = PDControl(...
301            kp_knee, kd_knee, knee_des_out, knee_pos, knee_vel, ...
302            kp_ankle, kd_ankle, ankle_des_out, ankle_pos, ankle_vel);
303       
304        else
305            knee_des_out = knee_pos;
306            ankle_des_out = ankle_pos;
307            [Knee_torque_command_swing,Ankle_torque_command_swing] = PDControl(...
308            kp_knee, kd_knee, knee_des_out, knee_pos, knee_vel, ...
309            kp_ankle, kd_ankle, ankle_des_out, ankle_pos, ankle_vel);
310        end
311        
312        %Smooth between stance and swing torques
313        Knee_torque_command = StanceGain*Knee_torque_command_stance + SwingGain*Knee_torque_command_swing;
314        Ankle_torque_command = StanceGain*Ankle_torque_command_stance + SwingGain*Ankle_torque_command_swing;
315        
316    else
317        IMU_LIVE = true;
318        %Use Setpoint PD control
319        knee_des_out = knee_des_in;
320        ankle_des_out = ankle_des_in;
321        [Knee_torque_command,Ankle_torque_command] = PDControl(...
322            kp_knee, kd_knee, knee_des_out, knee_pos, knee_vel, ...
323            kp_ankle, kd_ankle, ankle_des_out, ankle_pos, ankle_vel);        
324    end
325    
326    %% Virtual hard stops
327    u_stop = zeros(2,1);
328    if knee_stop_low < knee_stop_high && ankle_stop_low < ankle_stop_high
329        knee_limits = [knee_stop_low, knee_stop_high];  %deg, need to make sure its ordered
330        ankle_limits = [ankle_stop_low, ankle_stop_high];
331    else
332        knee_limits = single([KNEE_POS_MIN_LIM, KNEE_POS_MAX_LIM]);  %deg, need to make sure its ordered
333        ankle_limits = single([ANKLE_POS_MIN_LIM, ANKLE_POS_MAX_LIM]);
334    end
335
336    if knee_pos < knee_limits(1)
337        u_stop(1) = -kp_knee*(knee_pos - knee_limits(1)) - kd_knee*(knee_vel);
338    end
339    if knee_limits(2) < knee_pos 
340        u_stop(1) = -kp_knee*(knee_pos - knee_limits(2)) - kd_knee*(knee_vel);
341    end
342    if ankle_pos < ankle_limits(1)
343        u_stop(2) = -kp_ankle*(ankle_pos - ankle_limits(1)) - kd_ankle*(ankle_vel);
344    end
345    if ankle_limits(2) < ankle_pos
346        u_stop(2) = -kp_ankle*(ankle_pos - ankle_limits(2)) - kd_ankle*(ankle_vel);
347    end
348    U_STOP_K = u_stop(1);
349    U_STOP_A = u_stop(2);
350    Knee_torque_command = Knee_torque_command + u_stop(1);
351    Ankle_torque_command = Ankle_torque_command + u_stop(2);
352    
353    %% Saturate torque output
354    if max_torque>0
355        Knee_torque_command = Saturate(Knee_torque_command,-max_torque,max_torque);
356        Ankle_torque_command = Saturate(Ankle_torque_command,-max_torque,max_torque);
357    end
358        
359    %% Persistent/output variable management
360    Esys_integrate_out = Esys_integrate;
361    knee_pos_prev = knee_pos;
362    ankle_pos_prev = ankle_pos;
363    knee_vel_prev=knee_vel;
364    ankle_vel_prev=ankle_vel;
365    IMU_pitch_prev=IMU_pitch;
366    hip_vel_prev = hip_vel;
367    hip_pos_prev = hip_pos;
368    knee_joint_vel = single(knee_vel);
369    ankle_joint_vel = single(ankle_vel);
370    foot_contact = single(FootContact);
371    stance = single(Stance);
372    swing = single(Swing);
373    IMU_LIVE_OUT = single(IMU_LIVE);
374    PushOffOut = single(PushOff);
375end

378function [Knee_torque_command,Ankle_torque_command] = PDControl(kp_k, kd_k, q_kstar, q_k, qdot_k, kp_a, kd_a, q_astar, q_a, qdot_a)
379    %This control basically assumes that the desired joint angles are static i.e., the desired velocities are all zero.
380    Knee_torque_command = kp_k*(q_kstar - q_k) + kd_k*(-qdot_k);
381    Ankle_torque_command = kp_a*(q_astar - q_a) + kd_a*(-qdot_a);
382end

415function y = Saturate(x,L1,L2)
416    %Function to prevent the desired joint angles from changing to fast. 
417    %Works via saturation

419    y = max(y,min(L1,L2));
