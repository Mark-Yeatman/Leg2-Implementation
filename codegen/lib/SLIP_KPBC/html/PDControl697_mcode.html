 50        knee_pos_prev = 0;

 72    %Knee_torque_command = 0; 
 73    %Ankle_torque_command = 0; 
 74    %deltaL = single(0); %meters
 75    %Esys = single(0); %Joules
 76    %g = single(9.81); %m/s^2
 77    %pulse_length = single(10); %seconds
 78    %hip_pos = 0; 
 79    %t_out = 0; 
 80    %dt = 0;

 83    Mf = single(0.55 + 1.643130/2.205); %carbon fiber foot + mechanism, kg
 84    %lt = single(0.3733); %meters

 91    
 92    Knee_torque_command= single(0);
 93    Ankle_torque_command = single(0);
 94    knee_des_out = nan;
 95    ankle_des_out = nan;
 96    
 97    U_LIN_DAMP_K = single(0);
 98    U_LIN_DAMP_A = single(0);
 99    U_LIN_SPRING_K = single(0);
100    U_LIN_SPRING_A = single(0);
101    U_PBC_K = single(0);
102    U_PBC_A = single(0);

105    % Position/velocity hard limits
106    ANKLE_POS_MIN_LIM  = -35;
107    ANKLE_POS_MAX_LIM = 35;
108    ANKLE_VEL_MIN_LIM = -200; %deg/s
109    ANKLE_VEL_MAX_LIM = 200; %deg/s
110    KNEE_POS_MIN_LIM = 2;

113    KNEE_VEL_MAX_LIM = 400; %deg/s

118    a_knee=[-21.9472500000000,4.24525000000000,21.6720000000000,0;20.9295000000000,-6.97650000000000,7.71900000000000,0;15.9362500000000,-0.998250000000000,18.6640000000000,25.8830000000000;-59.8412500000000,13.6422500000000,64.8630000000000,0;-19.4405000000000,-1.40250000000000,38.4100000000000,-47.2960000000000;45.1070000000000,-7.15300000000000,0.456000000000000,0;1.42525000000000,0.0657500000000000,2.21000000000000,3.24500000000000];
119    a_ankle=[5.50050000000000,-0.880500000000000,-4.60000000000000,0;-8.98500000000000,3.77020000000000,5.26600000000000,4.65120000000000;0.812500000000000,-0.475000000000000,7.74100000000000,2.81250000000000;-2.19975000000000,0.320750000000000,9.62000000000000,0;-14.7575000000000,0.878500000000001,-0.745000000000000,-24.2440000000000;23.0932000000000,-3.91420000000000,-19.9240000000000,0;-29.8020000000000,9.93400000000000,-0.0560000000000000,0;0.703500000000000,-0.234500000000000,-0.525000000000000,0;-2.60850000000000,0.869500000000000,1.21400000000000,0;-0.729000000000000,0.0190000000000000,0.580000000000000,-1.34400000000000];
120    knee_ind=[1 139 400 530 720 848 976 1001];
121    ankle_ind=[1 61 205 330 440 550 653 846 898 977 1001];

124    %Assign joint positions
125    hip_pos = IMU_pitch;
126    knee_pos =  Knee_joint_position; 
127    ankle_pos = Ankle_joint_position;
128    
129    %Calculate joint velocities usindg exponential smoothing filter
130    %https://en.wikipedia.org/wiki/Exponential_smoothing
131    hip_vel = (1-vel_filter_coeff)*hip_vel_prev + vel_filter_coeff*(hip_pos-hip_pos_prev)/dt;
132    knee_vel = (1-vel_filter_coeff)*knee_vel_prev + vel_filter_coeff*(knee_pos-knee_pos_prev)/dt;
133    ankle_vel = (1-vel_filter_coeff)*ankle_vel_prev + vel_filter_coeff*(ankle_pos-ankle_pos_prev)/dt;
134    
135    %Human Leg as a Robot states
136    x = zeros(10,1); %x,y,-knee ankle, ankle angle
137    x(4) =  -knee_pos; %reverse knee sign convention of biomechanics versus biped modeling     
138    x(5) =  ankle_pos;
139
140    x(9) =  -knee_vel; %reverse knee sign convention of biomechanics versus biped modeling
141    x(10) =  ankle_vel;
142   
143    %Virtual linear spring from hip to foot, l1 is shank length, l2 is thigh length     
144    J = Spring_Jacobian_func(x,params);
145    L = Spring_Length_func(x,params);
146    deltaL = L - L0;
147    Ldot = Spring_vel_func(x,params);
148    
149    %Calculate system energy
150    %Ehip = 1/2*M*(Ldot^2) + (L-L0)*M*g; %kinetic and potential energy of slip model
151    Espring = 1/2*k_tilde*deltaL^2; %virtual spring potential energy
152    KE = 1/2*M*(Ldot^2);%KE_func(x,params);
153    %PE = PE_func(x,params);
154    Esys = Espring + KE;%Espring + PE + KE;
155    
156    %Phase Variable 
157    %s_a=clamp(1+(1-s_m)/(q_h_0-q_h_m)*(hip-q_h_0),0,1);
158    phase_var_out =  0.6 + 0.4*(hip_pos-Hip_Pos_Min)/(Hip_Pos_Max-Hip_Pos_Min);
159    phase_var_out = clamp(phase_var_out,0.6,1);
160    
161    %IMU State check
162    IMU_LIVE = abs(hip_pos_prev-hip_pos)<1;
163   
164    %PushOff = and(or((hip_pos < hip_thresh_st),PushOff),Stance); %PushOff is persistent, so this remembers when the hip_pos crosses and stays high until Stance is off
165    
166    %% Control logic
167    %Determine foot contact
168    if norm([Load_cell_x_force, Load_cell_y_force, Load_cell_z_force],2) > F_thresh
169        ForceCount = ForceCount+1;
170        if ForceCount > 2
171            FootContact = true;
172        else
173            FootContact = false;
174        end
175    else
176        FootContact = false;
177        ForceCount = 0;   
178    end
179             
180    StanceGain = 0;
181    SwingGain = 0;
182    t_hold = 0.3;
183    
184    %Phase State management, set autodetection or command stance/swing
185    if int8(Command_State) == int8(0)
186        %Determine stance vs swing, only allow phase state switching every t_hold seconds
187        if Stance && ~FootContact %The state needs to be switched
188            if t_hold < (time_in - t_switched_phase) %Check the last time we switched, if its too fast, wait
189               %Switch to SWING
190               t_switched_phase = time_in;
191               Swing = true;
192               Stance = false;
193               %reset the energy intergration variables
194               %Esys_integrate = 0;
195               %Edis = 0; 
196               u_pbc_knee_prev = 0;
197               u_pbc_ankle_prev = 0;
198            end
199        elseif Swing && FootContact %ditto, but for the other case
200            if t_hold < (time_in - t_switched_phase) 
201                %Switch to STANCE
202                t_switched_phase = time_in;
203                Swing = false;
204                Stance = true;
205            end
206        end
207        %Manage torque smoothing when switching phases
208        if Stance
209            t_norm = (time_in - t_switched_phase)/(t_hold*0.8);
210            StanceGain = min(t_norm,1);
211            SwingGain = 1-StanceGain; 
212        elseif Swing
213            t_norm = (time_in - t_switched_phase)/(t_hold*0.8);
214            SwingGain = min(t_norm,1);
215            StanceGain = 1-SwingGain; 
216        end
217    elseif int8(Command_State) == int8(1) %Stance
218        Stance = true;
219        Swing = false;
220        Hip_Pos_Max = 23;
221        Hip_Pos_Min = - 20;
222        StanceGain = 1;
223        SwingGain = 0;
224    elseif int8(Command_State) == int8(2) %Swing
225        Stance = false;
226        Swing = true;
227        Hip_Pos_Max = 23;
228        Hip_Pos_Min = - 20;
229        StanceGain = 0;
230        SwingGain = 1;
231    end
232
233    %Torque computation
234    if SLIP_ON       
235        Knee_torque_command_stance = 0;
236        Ankle_torque_command_stance = 0;

238        %Virtual linear spring
239        u_lin_spring = -k.*(deltaL).*J;

252%             Edis = Edis  + (knee_vel*u_lin_damp(4) + ankle_vel*u_lin_damp(5))*dt;
253%         end
254
255        if KPBC_ON %Also use energy tracking controller   
256            u_pbc = single(zeros(5,1));
257            
258%             if Joint_Bio_Sat 
259%                 %Joint level KPBC
260%                 u_pbc(4) = -pbc_gain_knee * knee_vel * (Esys_integrate - Eref - max(Edis,Edismax));
261%                 u_pbc(5) = -pbc_gain_ankle * ankle_vel * (Esys_integrate - Eref - max(Edis,Edismax));
262%                 
263%                 U_PBC_K = u_pbc(4); %THE SIGN NEEDS TO BE CHANGED IF USING SPRING VERSUS JOINT LEVEL BECAUSE OF BIO VS ROBOT KNEE CORDS
264%                 U_PBC_A = u_pbc(5);
265%                 
266%             else

269                
270                U_PBC_K = -u_pbc(4); %THE SIGN NEEDS TO BE CHANGED IF USING SPRING VERSUS JOINT LEVEL BECAUSE OF BIO VS ROBOT KNEE CORDS
271                U_PBC_A = u_pbc(5);
272                
273            %end
274            
275            %moving average filter and saturation laws on value and rate of
276            %torque
277         
278            %Absolute saturation
279            if KPBC_max_torque>0
280                U_PBC_K = Saturate(U_PBC_K,-KPBC_max_torque,KPBC_max_torque);
281                U_PBC_A = Saturate(U_PBC_A,-KPBC_max_torque,KPBC_max_torque);
282            end
283            
284            %Rate saturation
285            if KPBC_max_torque_rate>0
286                dk = (U_PBC_K-u_pbc_knee_prev)/dt;
287                da = (U_PBC_A-u_pbc_ankle_prev)/dt;

289                da = Saturate(da, -KPBC_max_torque_rate, KPBC_max_torque_rate);
290                U_PBC_K = dk*dt+u_pbc_knee_prev;
291                U_PBC_A = da*dt+u_pbc_ankle_prev;
292            end
293            
294            %exponential smoothing of torque
295            U_PBC_K = (1-KPBC_filter_coeff)*u_pbc_knee_prev + KPBC_filter_coeff*U_PBC_K;
296            U_PBC_A = (1-KPBC_filter_coeff)*u_pbc_ankle_prev + KPBC_filter_coeff*U_PBC_A;
297            
298            if Joint_Bio_Sat 
299                %Biomemetic power saturation
300                Pow_Knee = knee_vel*U_PBC_K ;
301                Pow_Ankle = ankle_vel*U_PBC_A;
302                if sign(Pow_Ankle) == -1
303                    U_PBC_A = 0;
304                end
305                if sign(Pow_Knee) == 1
306                    U_PBC_K = 0;
307                end
308            end
309
310            
311            %Use energy intergration scheme  
312            %Esys_integrate = Esys_integrate  + (knee_vel*U_PBC_K + ankle_vel*U_PBC_A)*dt;
313                     
314            %cancel damping term during KPBC
315            Knee_torque_command_stance = Knee_torque_command_stance + U_PBC_K - U_LIN_DAMP_K;
316            Ankle_torque_command_stance = Ankle_torque_command_stance + U_PBC_A - U_LIN_DAMP_A;
317        end
318        %% Swing
319        if IMU_LIVE
320            %Use hip pos as phase variable to index winters data
321            [knee_des_out ,~,ankle_des_out,~]=winterFit(phase_var_out*1000,a_knee,a_ankle,knee_ind,ankle_ind);
322
323            %follow the trjactory via PD controller
324            [Knee_torque_command_swing,Ankle_torque_command_swing] = PDControl(...
325            kp_knee, kd_knee, knee_des_out, knee_pos, knee_vel, ...
326            kp_ankle, kd_ankle, ankle_des_out, ankle_pos, ankle_vel);
327       
328        else
329            knee_des_out = knee_pos;
330            ankle_des_out = ankle_pos;
331            [Knee_torque_command_swing,Ankle_torque_command_swing] = PDControl(...
332            kp_knee, kd_knee, knee_des_out, knee_pos, knee_vel, ...
333            kp_ankle, kd_ankle, ankle_des_out, ankle_pos, ankle_vel);
334        end
335        
336        %Smooth between stance and swing torques
337        Knee_torque_command = StanceGain*Knee_torque_command_stance + SwingGain*Knee_torque_command_swing;
338        Ankle_torque_command = StanceGain*Ankle_torque_command_stance + SwingGain*Ankle_torque_command_swing;
339        
340    else
341        IMU_LIVE = true;
342        %Use Setpoint PD control
343        knee_des_out = knee_des_in;
344        ankle_des_out = ankle_des_in;
345        [Knee_torque_command,Ankle_torque_command] = PDControl(...
346            kp_knee, kd_knee, knee_des_out, knee_pos, knee_vel, ...
347            kp_ankle, kd_ankle, ankle_des_out, ankle_pos, ankle_vel);        
348    end
349    
350    %% Virtual hard stops
351    u_stop = zeros(2,1);
352    if knee_stop_low < knee_stop_high && ankle_stop_low < ankle_stop_high
353        knee_limits = [knee_stop_low, knee_stop_high];  %deg, need to make sure its ordered
354        ankle_limits = [ankle_stop_low, ankle_stop_high];
355    else
356        knee_limits = single([KNEE_POS_MIN_LIM, KNEE_POS_MAX_LIM]);  %deg, need to make sure its ordered
357        ankle_limits = single([ANKLE_POS_MIN_LIM, ANKLE_POS_MAX_LIM]);
358    end
359
360    if knee_pos < knee_limits(1)
361        u_stop(1) = -kp_knee*(knee_pos - knee_limits(1)) - kd_knee*(knee_vel);
362    end
363    if knee_limits(2) < knee_pos 
364        u_stop(1) = -kp_knee*(knee_pos - knee_limits(2)) - kd_knee*(knee_vel);
365    end
366    if ankle_pos < ankle_limits(1)
367        u_stop(2) = -kp_ankle*(ankle_pos - ankle_limits(1)) - kd_ankle*(ankle_vel);
368    end
369    if ankle_limits(2) < ankle_pos
370        u_stop(2) = -kp_ankle*(ankle_pos - ankle_limits(2)) - kd_ankle*(ankle_vel);
371    end
372    U_STOP_K = u_stop(1);
373    U_STOP_A = u_stop(2);
374    Knee_torque_command = Knee_torque_command + u_stop(1);
375    Ankle_torque_command = Ankle_torque_command + u_stop(2);
376    
377    %% Saturate torque output
378    if max_torque>0
379        Knee_torque_command = Saturate(Knee_torque_command,-max_torque,max_torque);
380        Ankle_torque_command = Saturate(Ankle_torque_command,-max_torque,max_torque);
381    end
382        
383    %% Persistent/output variable management
384    Esys_integrate_out = Esys_integrate;
385    knee_pos_prev = knee_pos;
386    ankle_pos_prev = ankle_pos;
387    knee_vel_prev=knee_vel;
388    ankle_vel_prev=ankle_vel;
389    IMU_pitch_prev=IMU_pitch;
390    hip_vel_prev = hip_vel;
391    hip_pos_prev = hip_pos;
392    u_pbc_knee_prev = U_PBC_K;
393    u_pbc_ankle_prev = U_PBC_A;
394    knee_joint_vel = single(knee_vel);
395    ankle_joint_vel = single(ankle_vel);
396    foot_contact = single(FootContact);
397    stance = single(Stance);
398    swing = single(Swing);
399    IMU_LIVE_OUT = single(IMU_LIVE);
400    PushOffOut = single(PushOff);
401end

404function [Knee_torque_command,Ankle_torque_command] = PDControl(kp_k, kd_k, q_kstar, q_k, qdot_k, kp_a, kd_a, q_astar, q_a, qdot_a)
405    %This control basically assumes that the desired joint angles are static i.e., the desired velocities are all zero.
406    Knee_torque_command = kp_k*(q_kstar - q_k) + kd_k*(-qdot_k);
407    Ankle_torque_command = kp_a*(q_astar - q_a) + kd_a*(-qdot_a);
408end

441function y = Saturate(x,L1,L2)
442    %Function to prevent the desired joint angles from changing to fast. 
443    %Works via saturation
444    y = min(x,max(L1,L2));
445    y = max(y,min(L1,L2));
446end
