 28        t_prev = time_in;

 32        ankle_pos_prev = 0;

 34        dknee_prev = 0;
 35        dankle_prev = 0;

 37    end

 40    %%

 61    knee_pos = Knee_motor_position;
 62    ankle_pos = Ankle_motor_position;
 63
 64    %Calculate joint velocities usindg weighted backwards difference
 65    dt = Iteration_time;
 66    t = Iteration_time + t;
 67    dknee = (1-filter_coeff)*dknee_prev + filter_coeff*(knee_pos-knee_pos_prev)/dt;
 68    dankle = (1-filter_coeff)*dankle_prev + filter_coeff*(ankle_pos-ankle_pos_prev)/dt;
 69    
 70    if FC %Use SLIP Embedding Controller
 71        %reverse ankle and shift because sign convention of biomechanics versus biped modeling
 72        ankle_pos = - ankle_pos;
 73        %[Knee_torque_command,Ankle_torque_command ,deltaL] = SLIPControl(ankle_pos, knee_pos, hip_pos, lf, la, ls, lt, k, L0);
 74        %Virtual linear spring from hip to foot
 75            fc = lf;
 76            %x should be 
 77            %l1 is shank length
 78            %l2 is thigh length
 79            %lf = 0; %at heel

 83            x(3) = ankle_pos;

 88            deltaL = L - L0;
 89            u = -k*(deltaL)*J;
 90            E = 1/2*k*deltaL^2;

 95%         if EnergyTracking
 96%             
 97%         end

101        [Knee_torque_command,Ankle_torque_command] = PDControl(...
102            kp_knee,kd_knee,knee_des_in,knee_pos,dknee, ...
103            kp_ankle,kd_ankle,ankle_des_in,ankle_pos,dankle);
104    end
105
106    %%
107    %Storing persistent variables for next iteration
108    knee_pos_prev = knee_pos;
109    ankle_pos_prev = ankle_pos;
110    t_out=t;
111    t_prev=time_in;
112    dknee_prev=dknee;
113    dankle_prev=dankle;
114    IMU_pitch_prev=hip_pos;
115end

119function [Knee_torque_command,Ankle_torque_command] = PDControl(kp_k, kd_k, q_kstar, q_k, qdot_k, kp_a, kd_a, q_astar, q_a, qdot_a)
120    %This control basically assumes that the desired joint angles are static i.e., the desired velocities are all zero.
121    Knee_torque_command = kp_k*(q_kstar-q_k) + kd_k*(-qdot_k);
122    Ankle_torque_command = kp_a*(q_astar-q_a) + kd_a*(-qdot_a);
123end

125function [Knee_torque_command,Ankle_torque_command,deltaL,E] = SLIPControl(ankle, knee, hip, fc, la, ls, lt, ks, L0)
126    %Virtual linear spring from hip to foot
127    %x should be 
128    %l1 is shank length
129    %l2 is thigh length
130    %lf = 0; %at heel
131    x = zeros(3,1);
132    x(1) = ankle+knee-hip;
133    x(2) = knee;
134    x(3) = ankle;
135
136    J = [0;(-1).*lt.*(2.*fc.*cosd(x(2)+x(3))+2.*ls.*sind(x(2))+la.*sind(x(2)+x(3))).*(4.*fc.^2+la.^2+4.*ls.^2+4.*lt.^2+8.*ls.*lt.*cosd(x(2))+4.*la.*ls.*cosd(x(3))+4.*la.*lt.*cosd(x(2)+x(3))+(-8).*fc.*ls.*sind(x(3))+(-8).*fc.*lt.*sind(x(2)+x(3))).^(-1/2);(-1).*(4.*fc.^2+la.^2+4.*ls.^2+4.*lt.^2+8.*ls.*lt.*cosd(x(2))+4.*la.*ls.*cosd(x(3))+4.*la.*lt.*cosd(x(2)+x(3))+(-8).*fc.*ls.*sind(x(3))+(-8).*fc.*lt.*sind(x(2)+x(3))).^(-1/2).*(2.*fc.*ls.*cosd(x(3))+2.*fc.*lt.*cosd(x(2)+x(3))+la.*ls.*sind(x(3))+la.*lt.*sind(x(2)+x(3)))];
137    L  = (fc.^2+(1/4).*la.^2+ls.^2+lt.^2+2.*ls.*lt.*cosd(x(2))+la.*ls.*cosd(x(3))+la.*lt.*cosd(x(2)+x(3))+(-2).*fc.*ls.*sind(x(3))+(-2).*fc.*lt.*sind(x(2)+x(3))).^(1/2);
138    deltaL = L - L0;
139    u = -ks*(deltaL)*J;
140    E = 1/2*ks*deltaL^2;
